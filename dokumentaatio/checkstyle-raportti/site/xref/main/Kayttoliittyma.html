68      /**
69       * Metodi asettaa layoutiksi gridBagLayoutin, johon tulee kolme JPanel komponenttia vertikaalisesti.
70       * 
71       * @param container 
72       */
73      private void luoKomponentit(Container container) {
74          container.setLayout(new GridBagLayout());
75          GridBagConstraints c = new GridBagConstraints();
76          c.fill = GridBagConstraints.BOTH;
77  
78          JPanel ohjeet = luoOhjeboksi();
79          c.fill = GridBagConstraints.BOTH;
80          c.gridx = 0;
81          c.gridy = 0;
82          c.gridheight = 1;
83          container.add(ohjeet, c);
84  
85          JPanel nopat = luoNopat();
86          c.fill = GridBagConstraints.BOTH;
87          c.gridx = 0;
88          c.gridy = 1;
89          c.gridheight = 1;
90          container.add(nopat, c);
91  
92          JPanel taulukko = luoTulostaulukko();
93          c.fill = GridBagConstraints.BOTH;
94          c.gridx = 0;
95          c.gridy = 2;
96          c.gridheight = 4;
97          container.add(taulukko, c);
98      }

199     /**
200      * Metodi selvittää mikä nappi aiheutti ActionEventin ja reagoi sen mukaisesti.
201      * 
202      * @param e ActienEvent e
203      * @see main.Kayttoliittyma#reagoiHeittonappiin(javax.swing.JButton) 
204      * @see main.Kayttoliittyma#reagoiNoppaan(javax.swing.JToggleButton) 
205      * @see main.Kayttoliittyma#reagoiSarakkeeseen(javax.swing.JToggleButton) 
206      */
207     @Override
208     public void actionPerformed(ActionEvent e) {
209         try {
210             JToggleButton button = (JToggleButton) e.getSource();
211 
212             if (this.nopat.contains(button)) {
213                 reagoiNoppaan(button);
214             }
215 
216             if (this.sarakkeet.contains(button)) {
217                 reagoiSarakkeeseen(button);
218             }
219 
220         } catch (Exception exptn) {
221             JButton button = (JButton) e.getSource();
222 
223             reagoiHeittonappiin(button);
224         }
225     }

228      * Kun jotakin nopista painetaan, metodi välittää painetun nopan indeksin pelilogiikalle.

241     /**
242      * Kun jotakin Tulostaulukon sarakkeista painetaan, metodi välittää painetun napin tekstin pelilogiikalle.
243      * Pelilogiikka selvittää voiko pisteet tallentaa kyseiseen sarakkeeseen ja palauttaa -1 jos ei voi. Muussa tapauksessa pelilogiikka palauttaa pistemäärän, joka tallennetaan sarakkeen viereiseen JLabel:iin.
244      * Jos pisteet tallennettiin, muutetaan heittonapin otsikko, jotta nopat voidaan heittää uudestaan.
245      * 
246      * @param button nappi, jota klikattiin
247      * @see yatzy.Pelilogiikka#laskeSarakkeenPisteet(int) 
248      * @see main.Kayttoliittyma#paivitaSummaJaValisumma() 
249      * @see main.Kayttoliittyma#lukitseTaulukko() 
250      */
251     public void reagoiSarakkeeseen(JToggleButton button) {
252 
253         int sarake = selvitaSarake(button);
254         int pointsit = this.logiikka.laskeSarakkeenPisteet(sarake);
255         
256         if (pointsit == -1) {
257             return;
258         }
259         
260         this.pisteet.get(sarake).setText(String.valueOf(pointsit));
261 
262         paivitaSummaJaValisumma();
263         lukitseTaulukko();
264         this.heittaja.setText("Aloita peli");
265     }
266 
267     /**
268      * Metodi selvittää painetun napin indeksin ja palauttaa sitä vastaavan JLabel:in.
269      * 
270      * @param button painettu nappi
271      * @return nappia vastaava JLabel
272      */
273     public int selvitaSarake(JToggleButton button) {
274         for (int i = 0; i < this.sarakkeet.size(); i++) {
275             if (this.sarakkeet.get(i).equals(button)) {
276                 return i;
277             }
278         }
279         return -1;
280     }

293     /**
294      * REagoi heittonapin painallukseen heittämällä nopat ja päivittämällä kuvat ja napin tekstin.
295      * 
296      * @param button painettu nappi
297      */
298     public void reagoiHeittonappiin(JButton button) {
299         if (button.getText() == "Aloita peli") {
300             heitaKaikkiNopat();
301             paivitaNopat();
302             button.setText("Heitä valitsemani nopat uudestaan (1. kerta)");
303         } else if (button.getText() == "Heitä valitsemani nopat uudestaan (1. kerta)") {
304             heitaValitutNopat();
305             paivitaNopat();
306             button.setText("Heitä valitsemani nopat uudestaan (2. kerta)");
307         } else if (button.getText() == "Heitä valitsemani nopat uudestaan (2. kerta)") {
308             heitaValitutNopat();
309             paivitaNopat();
310             button.setText("Kirjaa tulos");
311             vapautaTaulukko();
312         }
313     }

329     /**
330      * Päivittää noppanappeihin oikeaa silmälukua esittävät kuvat ja vapauttaa nopat.
331      */
332     public void paivitaNopat() {
333         int[] silmaluvut = this.logiikka.getSilmaluvut();
334         for (int i = 0; i < 5; i++) {
335             int silmaluku = silmaluvut[i];
336             String kuvanOsoite = "src/noppa" + silmaluku + ".png";
337             this.nopat.get(i).setIcon(new ImageIcon(kuvanOsoite));
338         }
339 
340         for (JToggleButton noppa : nopat) {
341             if (noppa.isSelected()) {
342                 noppa.setSelected(false);
343             }
344         }
345     }
