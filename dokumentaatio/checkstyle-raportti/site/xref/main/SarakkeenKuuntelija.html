1   package main;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import java.util.*;
6   import javax.swing.*;
7   import yatzy.Pelilogiikka;
8   
9   /**
10   * SarakkeenKuuntelija reagoi taulukon sarakkeen klikkauksiin.
11   * 
12   * @author iiris
13   */
14  public class SarakkeenKuuntelija implements ActionListener {
15      private Pelilogiikka logiikka;
16      private ArrayList<JToggleButton> sarakkeet;
17      private ArrayList<JLabel> pisteet;
18      private Kayttoliittyma k;
19      
20      /**
21       * Konstruktori linkkaa kuuntelijan pelilogiikkaan ja käyttöliittymään sekä käyttöliittymän sarake- ja pistelistoihin.
22       * @param l pelilogiikka
23       * @param k käyttöliittymä
24       * @param sarakkeet lista sarakenappuloista
25       * @param pisteet lista JLabeleita, joihin pisteen kirjataan
26       */
27      public SarakkeenKuuntelija(Pelilogiikka l, Kayttoliittyma k, ArrayList<JToggleButton> sarakkeet, ArrayList<JLabel> pisteet) {
28          this.logiikka = l;
29          this.pisteet = pisteet;
30          this.sarakkeet = sarakkeet;
31          this.k = k;
32      }
33  
34      /**
35       * Metodi reagoi sarakkeen klikkaukseen. Se selvittää mitä saraketta klikattiin, pyytää logiikkaa laskemaan pisteet ja kirjoittaa pisteet nappulaa vastaavaan JLabeliin.
36       * Lopuksi metodi päivittää välisumman ja summan, lukitsee taulukon ja muuttaa ohjekentän tekstin.
37       * 
38       * @param e ActionEvent
39       */
40      @Override
41      public void actionPerformed(ActionEvent e) {
42          JToggleButton button = (JToggleButton) e.getSource();
43          
44          int sarake = selvitaSarake(button);
45          int pointsit = this.logiikka.laskeSarakkeenPisteet(sarake);
46          
47          if (pointsit == -1) {
48              return;
49          }
50          
51          this.pisteet.get(sarake).setText(String.valueOf(pointsit));
52  
53          paivitaSummaJaValisumma();
54          k.lukitseTaulukko();
55          k.setLabel("Aloita seuraava kierros");
56      }
57      
58      /**
59       * Metodi selvittää ja palauttaa painetun napin indeksin listassa.
60       * 
61       * @param button painettu nappi
62       * @return napin indeksi listassa
63       */
64      public int selvitaSarake(JToggleButton button) {
65          for (int i = 0; i < this.sarakkeet.size(); i++) {
66              if (this.sarakkeet.get(i).equals(button)) {
67                  return i;
68              }
69          }
70          return -1;
71      }
72      
73      /**
74       * Hakee taulukkoon pelilogiikasta nykyisen summan ja välisumman.
75       */
76      public void paivitaSummaJaValisumma() {
77          int valisumma = this.logiikka.getValisumma();
78          int summa = this.logiikka.getSumma();
79          
80          this.pisteet.get(6).setText(String.valueOf(valisumma));
81          this.pisteet.get(16).setText(String.valueOf(summa));
82      }
83  }
