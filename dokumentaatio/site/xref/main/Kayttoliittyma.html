170     @Override
171     public void actionPerformed(ActionEvent e) {
172         try {
173             JToggleButton button = (JToggleButton) e.getSource();
174 
175             if (this.nopat.contains(button)) {
176                 reagoiNoppaan(button);
177             }
178 
179             if (this.sarakkeet.contains(button)) {
180                 reagoiSarakkeeseen(button);
181             }
182 
183         } catch (Exception exptn) {
184             JButton button = (JButton) e.getSource();
185 
186             reagoiHeittonappiin(button);
187         }
188     }

190     public void reagoiNoppaan(JToggleButton button) {
191         for (int i = 0; i < 5; i++) {
192             if (this.nopat.get(i).equals(button)) {
193                 this.logiikka.valitseNoppa(i);
194             }
195         }
196     }

198     public void reagoiSarakkeeseen(JToggleButton button) {
199         if (button.getText().equals("VÄLISUMMA") || button.getText().equals("SUMMA")) {
200             return;
201         }
202 
203         JLabel sarake = selvitaSarake(button);
204 
205         int pisteet = this.logiikka.laskeSarakkeenPisteet(button.getText());
206         sarake.setText(String.valueOf(pisteet));
207         sarake.setEnabled(false);
208 
209         paivitaSummaJaValisumma();
210         lukitseTaulukko();
211         this.heittaja.setText("Heitä nopat");
212     }
213 
214     public JLabel selvitaSarake(JToggleButton button) {
215         for (int i = 0; i < this.sarakkeet.size(); i++) {
216             if (this.sarakkeet.get(i).equals(button)) {
217                 return this.pisteet.get(i);
218             }
219         }
220 
221         return null;
222     }

228     public void reagoiHeittonappiin(JButton button) {
229         if (button.getText() == "Heitä nopat") {
230             heitaKaikkiNopat();
231             paivitaNopat();
232             button.setText("Heitä valitsemani nopat uudestaan (1. kerta)");
233         } else if (button.getText() == "Heitä valitsemani nopat uudestaan (1. kerta)") {
234             heitaValitutNopat();
235             paivitaNopat();
236             button.setText("Heitä valitsemani nopat uudestaan (2. kerta)");
237         } else if (button.getText() == "Heitä valitsemani nopat uudestaan (2. kerta)") {
238             heitaValitutNopat();
239             paivitaNopat();
240             button.setText("Kirjaa tulos");
241             vapautaTaulukko();
242         }
243     }

253     public void paivitaNopat() {
254         int[] silmaluvut = this.logiikka.getSilmaluvut();
255         for (int i = 0; i < 5; i++) {
256             int silmaluku = silmaluvut[i];
257             String kuvanOsoite = "src/noppa" + silmaluku + ".png";
258             this.nopat.get(i).setIcon(new ImageIcon(kuvanOsoite));
259         }
260 
261         for (JToggleButton noppa : nopat) {
262             if (noppa.isSelected()) {
263                 noppa.setSelected(false);
264             }
265         }
266     }
