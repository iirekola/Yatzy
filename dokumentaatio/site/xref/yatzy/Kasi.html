1   package yatzy;
2   
3   /**
4    * Käsi koostuu viidestä nopasta ja tarjoaa metodit noppien käsittelyyn. Käsi tallentaa noppien kulloisenkin tilan.
5    */
6   import java.util.ArrayList;
7   
8   public class Kasi {
9   
10      private Noppa[] nopat;
11  
12      /**
13       * Konstruktori luo taulukon, jossa on paikat viidelle noppaoliolle. 
14       *
15       * @see yatzy.Kasi#luoNopat()
16       */
17      public Kasi() {
18          this.nopat = new Noppa[5];
19          luoNopat();
20      }
21  
22      /**
23       * luoNopat() luo kädelle viisi Noppa-oliota.
24       *
25       * @see yatzy.Noppa
26       */
27      private void luoNopat() {
28          for (int i = 0; i < 5; i++) {
29              nopat[i] = new Noppa();
30          }
31      }
32  
33      /**
34       * getNopat() palauttaa kädessä olevat Nopat.
35       * 
36       * @return viiden kokoinen Noppa[]-taulukko
37       */
38      public Noppa[] getNopat() {
39          return this.nopat;
40      }
41  
42      /**
43       * heitaNoppa(Noppa n) kutsuu nopan n metodia heitaNoppaa.
44       * 
45       * @param n Noppa-olio, jonka arvo halutaan arpoa uudelleen.
46       */
47      public void heitaNoppa(Noppa n) {
48          n.heitaNoppaa();
49      }
50  
51      /**
52       * heitaValitutNopat() heittaa nopat, jotka ovat valittuina.
53       * 
54       * @see yatzy.Noppa#getValittu() 
55       * @see yatzy.Kasi#heitaNoppa(yatzy.Noppa)
56       */
57      public void heitaValitutNopat() {
58          for (Noppa noppa : nopat) {
59              if (noppa.getValittu()) {
60                  heitaNoppa(noppa);
61              }
62          }
63      }
64  
65      /**
66       * heitaKaikkiNopat() heittää kaikki käden nopat.
67       * 
68       * @see yatzy.Kasi#heitaNoppa(yatzy.Noppa)
69       */
70      public void heitaKaikkiNopat() {
71          for (Noppa noppa : nopat) {
72              heitaNoppa(noppa);
73          }
74      }
75  
76      /**
77       * valitseNoppa(int i) valitsee nopan taulukon indeksistä i.
78       * 
79       * @param i valittavan nopan indeksi taulukossa
80       * @see yatzy.Noppa#setValittu(boolean) 
81       */
82      public void valitseNoppa(int i) {
83          this.nopat[i].setValittu(true);
84      }
85  
86      /**
87       * vapautaNoppa(int i) vapauttaa nopan taulukon indeksistä i.
88       * 
89       * @param i vapautettavan nopan indeksi taulukossa
90       * @see yatzy.Noppa#setValittu(boolean) 
91       */
92      public void vapautaNoppa(int i) {
93          this.nopat[i].setValittu(false);
94      }
95  
96      /**
97       * vapautaKaikkiNopat() vapauttaa kaikki kädessä olevat nopat.
98       * 
99       * @see yatzy.Noppa#setValittu(boolean) 
100      */
101     public void vapautaKaikkiNopat() {
102         for (Noppa noppa : nopat) {
103             noppa.setValittu(false);
104 
105         }
106     }
107 
108     /**
109      * getSilmaluvut() palauttaa käden noppien tämänhetkiset silmäluvut int[] -taulukkona.
110      * 
111      * @return viiden kokoinen int[]-taulukko, jossa noppien silmäluvut
112      * @see yatzy.Noppa#getArvo() 
113      */
114     public int[] getSilmaluvut() {
115         int[] arvot = new int[5];
116         for (int i = 0; i < 5; i++) {
117             arvot[i] = this.nopat[i].getArvo();
118         }
119 
120         return arvot;
121     }
122 
123 }
